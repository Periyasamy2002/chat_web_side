#!/usr/bin/env python
"""
Test to verify short messages are accepted by the backend
"""

assert __name__ == '__main__', "Run this as: python manage.py shell < this_file"

from django.test import Client
from chatapp.models import Group, Message, AnonymousUser

print("=" * 80)
print("TEST: Short Message Sending")
print("=" * 80)

# Create test client
client = Client()

# Create or get test group  
group, created = Group.objects.get_or_create(code='SHORTMSG', defaults={'name': 'Short Message Test'})
print(f"\nGroup: {group.code} (created={created})")

# Access group to create session
print("\nStep 1: Visiting /group/SHORTMSG/ to create session...")
response = client.get(f'/group/{group.code}/')
print(f"  Status: {response.status_code}")

# Get CSRF token
csrf_token = response.cookies.get('csrftoken', '')
print(f"  CSRF Token: {'Found' if csrf_token else 'NOT FOUND'}")

# Store current message count
before_count = Message.objects.filter(group=group).count()

# Test data
test_cases = [
    ('hi', 'short 2-char message'),
    ('h', 'single character'),
    ('OK', 'two characters'),
    ('hello', 'normal word'),
]

print(f"\nStep 2: Testing different short messages...")
print("-" * 80)

results = []

for message_text, description in test_cases:
    print(f"\nTest: '{message_text}' ({description})")
    print(f"  Length: {len(message_text)} characters")
    
    # Send message  
    response = client.post(f'/group/{group.code}/send-message/', {
        'message': message_text
    }, HTTP_X_CSRFTOKEN=csrf_token, HTTP_X_REQUESTED_WITH='XMLHttpRequest')
    
    print(f"  HTTP Status: {response.status_code}")
    
    # Parse response
    try:
        data = response.json()
        success = data.get('success', False)
        print(f"  Response Success: {success}")
        
        if not success:
            print(f"  Error: {data.get('error', 'Unknown error')}")
            results.append((message_text, False))
        else:
            # Check if message was created
            messages = Message.objects.filter(group=group, content=message_text).order_by('-id')
            if messages.exists():
                msg = messages.first()
                print(f"  ✓ Message created (ID: {msg.id})")
                print(f"    - Content: '{msg.content}'")
                print(f"    - User: {msg.user_name}")
                results.append((message_text, True))
            else:
                print(f"  ✗ Message NOT in database")
                results.append((message_text, False))
    except Exception as e:
        print(f"  Exception: {str(e)}")
        results.append((message_text, False))

# Final summary
print("\n" + "=" * 80)
print("SUMMARY")
print("=" * 80)

for message_text, success in results:
    status = "✓ PASS" if success else "✗ FAIL"
    print(f"{status}: '{message_text}'")

# Check database
print(f"\nMessages in '{group.code}' group:")
all_messages = Message.objects.filter(group=group).order_by('-id')
for msg in all_messages:
    print(f"  - ID {msg.id}: '{msg.content}' (user: {msg.user_name}, type: {msg.message_type})")

print(f"\nTotal messages: {all_messages.count()}")
print(f"Messages added: {all_messages.count() - before_count}")

print("\n" + "=" * 80)
all_pass = all(success for _, success in results)
print(f"OVERALL RESULT: {'✓ ALL TESTS PASSED' if all_pass else '✗ SOME TESTS FAILED'}")
print("=" * 80)
